```typescript
// Default: Groq Llama 3.3 70B
if (task.complexity === "planning") return "groq";
if (task.tokens > 32_000) return "gemini" /* 3.1 Pro free tier 1M context */;
if (task.kind === "classification") return "groq";
if (task.kind === "verification") {
  if (verifierCallsThisSession < sessionCap * 0.1) return "sonnet"; // ≤10%

}
if (task.kind === "vision") return "gemini";
if (task.durationEstimate > 30 * 60 * 1000) return "gemini"; // long-horizon, 1M context
if (rateLimited(preferred)) return nextInFallbackChain(preferred);
```
